import React from "react"

import Layout from "../components/Layout"

import Services from "../components/Services"
import Jobs from "../components/Jobs"
import Projects from "../components/Projects"
import Blogs from "../components/Blogs"
import SEO from "../components/SEO"




export default ({data}) => {
  // console.log(data)
  const {
    allStrapiProjects: {nodes:projects},
    allStrapiBlogs: {nodes:blogs}
  } = data;


  return (
    <Layout>
      <SEO title="Home" description="this is our home page"/>

        projects={projects}

    )
}

At the beginning of this post it was mentioned that SEO plays an extremely important role in spreading the word of your presence. Search Engin Optimization is a plug in that is offered by Gatsby and it would be worth mentioning how to build it and share it on Twitter via Twitter card. The plug in responsible for the SEO can be found here, documentation on how to use it is very intuitive and will allow to create a small SEO component in seconds. Belwo the file SEO.js that specifically does that for the resume application:

js
import React from 'react'
import { Helmet } from 'react-helmet'
import { useStaticQuery, graphql } from 'gatsby'


const query = graphql`
  {
    site {
      siteMetadata {
        author
        siteDesc: description
        image
        siteUrl
        siteTitle: title
        twitterUsername
      }
    }
  }
`
const SEO = ({ title, description }) => {
  // lets pull out the data from the graphql query
  // we will use a staticQuery
  const { site } = useStaticQuery(query)
  const {
    siteDesc,
    siteTitle,
    siteUrl,
    image,
    twitterUsername,
  } = site.siteMetadata


  return (
    <Helmet htmlAttributes={{ lang: 'en' }} title={`${title} | ${siteTitle}`}>
      <meta name="description" content={description || siteDesc} />
      <meta name="image" content={image} />
      {/* twitter cards */}
      <meta name="twitter:card" content="summary_large_image" />
      <meta name="twitter:creator" content={twitterUsername} />
      <meta name="twitter:title" content={siteTitle} />
      <meta name="twitter:description" content={siteDesc} />
      <meta name="twitter:image" content={`${siteUrl} ${image}`} />
    </Helmet>
  )
}
export default SEO
